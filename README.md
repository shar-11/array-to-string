# array-to-string
    Class chararraytostring
    {
       Public static void main(string args[])
       {
         Char[]ch={'g','o','o','d','m','o','r','n','i','n','g'};
         String str=new string(ch);
         System.out.println(str);
         String str2=string.valueof(ch);
         System.out.println(str2);
       }
    }
